enum DXGI_SWAP_CHAIN_FLAG {
  DXGI_SWAP_CHAIN_FLAG_NONPREROTATED                           // 자동 이미지 회전을 끔. (전체화면 모드에서만 유효)
  DXGI_SWAP_CHAIN_FLAG_ALLOW_MODE_SWITCH                       // IDXGISwapChain::ResizeTarget을 호출하여 모드를 전환할 수 있게 함.

  DXGI_SWAP_CHAIN_FLAG_RESTRICT_SHARED_RESOURCE_DRIVER         // 공유 리소스를 보호해야함을 나타낼때 사용.
  DXGI_SWAP_CHAIN_FLAG_DISPLAY_ONLY                            // 창의 내용을 다른 디스플레이로 복제하지 못하게 함.

  DXGI_SWAP_CHAIN_FLAG_FOREGROUND_LAYER                        // 다중 표면 렌더링을 위해 포어 그라운드 레이어에 스왑체인을 생성하려면 선택
  DXGI_SWAP_CHAIN_FLAG_FULLSCREEN_VIDEO                        // 전체 화면 비디오를 위한 스왑체인을 만드려면 선택
  DXGI_SWAP_CHAIN_FLAG_YUV_VIDEO                               // YUV 비디오의 스왑체인을 만드려면 선택
  DXGI_SWAP_CHAIN_FLAG_HW_PROTECTED                            // 모든 리소스가 하드웨어에 의해 보호 될 수 있도록 스왑체인을 만들어야함을 표시.
  DXGI_SWAP_CHAIN_FLAG_ALLOW_TEARING                           // 
  DXGI_SWAP_CHAIN_FLAG_RESTRICTED_TO_ALL_HOLOGRAPHIC_DISPLAYS  //
} ;
